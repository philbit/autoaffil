% test-superaftercomma.tex
% The [superaftercomma] option: superscripts appear after the comma that
% separates author names (revtex4-2 / Physical Review style).
\documentclass{article}
\usepackage[superaftercomma]{autoaffil}

\autoauthor{Alice Anderson}
  \autoaffil{Institute One}
\autoauthor{Bob Brown}
  \autoaffil{Institute Two}
  \autoaffil{Institute One}
\autoauthor{Carol Chen}
  \autoaffil{Institute Two}

\title{Test: superaftercomma}
\date{2026-01-01}

\begin{document}
\maketitle
\section{Body}
Superscripts follow the comma, not precede it.
\end{document}
